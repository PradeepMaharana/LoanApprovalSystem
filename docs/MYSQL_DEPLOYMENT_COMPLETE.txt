🔍 SAMPLE DATA QUERIES
════════════════════════════════════════════════════════════════════════════════

View All Applicants:
mysql> SELECT * FROM applicants LIMIT 10;

Get Approved Applicants:
mysql> SELECT * FROM loan_applications WHERE application_status = 'APPROVED' LIMIT 10;

Find High-Risk Applicants:
mysql> SELECT * FROM loan_applications WHERE risk_score < 40 LIMIT 10;

View by Location:
mysql> SELECT location, COUNT(*) FROM applicants GROUP BY location ORDER BY COUNT(*) DESC;

Get Statistics:
mysql> SELECT
    COUNT(*) as total,
    ROUND(AVG(age), 1) as avg_age,
    ROUND(AVG(income), 0) as avg_income,
    ROUND(AVG(credit_score), 0) as avg_credit
FROM applicants a
JOIN loan_applications l ON a.applicant_id = l.applicant_id;
